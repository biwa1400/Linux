1. install samba
2. create a share folder and change mode to 777
	mkdir share
    chmod 777 share
3. edit config file '/etc/samba/smb.conf'
	vim /etc/samba/smb.conf
	-----
		[PiShare]
		 comment=Raspberry Pi Share
		 path=/home/pi/share
		 browseable=Yes
		 writeable=Yes
		 only guest=no
		 create mask=0777
		 directory mask=0777
		 public=no
	-----
4. restart samba server
	 /etc/init.d/samba retsart
